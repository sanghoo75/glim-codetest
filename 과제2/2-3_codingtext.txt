#include <string>
#include <vector>

using namespace std;

int solution(vector<vector<int>> dots) {
    
    //제한사항 dots의 길이 = 4
    if (dots.size() != 4) return -1000;
    
    int answer = 0;
    
    //좌표계는 센서좌표계.
    int nFind = 0;
    int nAddMin = 999;
    int nAddMax = -999;
    int nSubMin = 999;
    int nSubMax = -999;
    
    int nFindLT = -1; //X+Y 가장 작은값
    int nFindRB = -1; //X+Y 가장 큰값
    int nFindRT = -1; //X-Y 가장 큰값
    int nFindLB = -1; //X-Y 가장 작은값
    
    const int x = 0;
    const int y = 1;
    
    for (int i = 0; i < dots.size(); ++i) {
        //제한사항 dots의 원소의 길이 = 2
        if (dots[i].size() != 2) return -1001;
        
        //제한사항 -256 < dots[i]의 원소 < 256
        if (dots[i][x] <= -256 || dots[i][x] >= 256 ) return -1002;
        if (dots[i][y] <= -256 || dots[i][y] >= 256 ) return -1003;
        
        int nAdd = dots[i][x] + dots[i][y];
        int nSub = dots[i][x] - dots[i][y];
        
        if (nAddMin > nAdd){
            nAddMin = nAdd;
            nFindLT = i;
        }

        if (nAddMax < nAdd){
            nAddMax = nAdd;
            nFindRB = i;
        }

        if (nSubMax < nSub){
            nSubMax = nSub;
            nFindRT = i;
        }
        if (nSubMin > nSub){
            nSubMin = nSub;
            nFindLB = i;
        }
    }
    
    //좌상/우상/ 좌하/ 우하 못찾았을 경우
    if (nFindLT < 0) return -2000;
    if (nFindRT < 0) return -2001;
    if (nFindLB < 0) return -2002;
    if (nFindRB < 0) return -2003;
    
    //넓이계산
    int nWidth = abs(dots[nFindRB][x] - dots[nFindLT][x]);
    int nHeight = abs(dots[nFindRB][y] - dots[nFindLT][y]);

    answer = nWidth * nHeight;
    
    return answer;
}
