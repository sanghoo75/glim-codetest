#include <string>
#include <vector>

using namespace std;

int solution(int n, int k) {
    int answer = 0;
    
    //제한사항 0 < n < 1,000
    if (n <= 0) return -1000;
    if (n >= 1000) return -1001;
    
    //제한사항 n / 10 ≤ k < 1,000
    if (k < n/10) return -1002;
    if (k >= 1000) return -1003;
    
    int nFoodPrice = n * 12000;
    int nDrinkPrice = k * 2000;
    int nDrinkSerive = -((int)(n / 10) * 2000);
    answer = nFoodPrice + nDrinkPrice + nDrinkSerive;
    return answer;
}
